def self.search_cardapio(data,campus)
    
    return if data.nil? || data.empty?
    return if campus.nil? || campus.empty?

    @parameter_data = data
    @parameter_campus  = campus
    Cardapio.where('data LIKE ? AND campus LIKE ?', "%#{@parameter_data}%", "%#{@parameter_campus}%").first
end
